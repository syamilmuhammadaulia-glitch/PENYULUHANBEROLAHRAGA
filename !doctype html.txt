  <script>
    // Simple SPA navigation
    const pages = document.querySelectorAll('.page')
    const navLinks = document.querySelectorAll('nav a')
    function showSection(id) {
      pages.forEach(p => p.style.display = p.id === id ? '' : 'none')
      navLinks.forEach(a => a.classList.toggle('active', a.dataset.target === id))
      window.location.hash = id
    }
    document.getElementById('mainNav').addEventListener('click', e => {
      if (e.target.matches('a')) {
        e.preventDefault()
        showSection(e.target.dataset.target)
      }
    })
    // init
    const initial = (location.hash && location.hash.slice(1)) || 'home'
    showSection(initial)

    // QUIZ data and logic
    const quizData = [
      { q: 'Berapa lama minimal berolahraga ringan tiap hari untuk pemula', choices: ['5 menit','15 menit','2 jam'], a: 1 },
      { q: 'Mana yang termasuk olahraga kardio', choices: ['Lari','Angkat beban','Yoga'], a: 0 },
      { q: 'Sebelum latihan sebaiknya melakukan', choices: ['Pemanasan','Makan berat','Tidur siang'], a: 0 },
      { q: 'Olahraga bantu meningkatkan', choices: ['Kekuatan otot','Warna rambut','Tinggi badan langsung'], a: 0 },
      { q: 'Saat merasa pusing saat olahraga, Anda harus', choices: ['Lanjut terus','Berhenti dan istirahat','Minum kopi'], a: 1 }
    ]

    let qIndex = 0
    const qtext = document.getElementById('qtext')
    const answers = document.getElementById('answers')
    const prevBtn = document.getElementById('prevBtn')
    const nextBtn = document.getElementById('nextBtn')
    const scoreEl = document.getElementById('score')
    const userAnswers = Array(quizData.length).fill(null)

    function renderQuestion() {
      const item = quizData[qIndex]
      qtext.textContent = `${qIndex + 1}. ${item.q}`
      answers.innerHTML = ''
      item.choices.forEach((c, i) => {
        const b = document.createElement('button')
        b.className = 'btn btn-outline'
        b.textContent = c
        b.onclick = () => {
          userAnswers[qIndex] = i
          renderQuestion()
          updateScoreText()
        }
        if (userAnswers[qIndex] === i) b.style.border = '2px solid #19a974'
        answers.appendChild(b)
      })
      prevBtn.disabled = qIndex === 0
      nextBtn.textContent = qIndex === quizData.length - 1 ? 'Selesai' : 'Selanjutnya'
    }

    function updateScoreText() {
      const answered = userAnswers.filter(x => x !== null).length
      scoreEl.textContent = `Terjawab ${answered} dari ${quizData.length}`
    }

    prevBtn.addEventListener('click', () => { if (qIndex > 0) { qIndex--; renderQuestion() } })
    nextBtn.addEventListener('click', () => {
      if (qIndex < quizData.length - 1) {
        qIndex++; renderQuestion()
      } else {
        // finish
        let correct = 0
        userAnswers.forEach((ans, i) => { if (ans === quizData[i].a) correct++ })
        scoreEl.textContent = `Skor: ${correct} / ${quizData.length}`
        alert('Kuis selesai. Skor: ' + correct + ' dari ' + quizData.length)
      }
    })

    renderQuestion()
    updateScoreText()

    // IMAGE GUESS game
    const imageItems = [
      { icon: '‚öΩ', name: 'Sepak bola' },
      { icon: 'üëü', name: 'Sepatu' },
      { icon: 'ü™¢', name: 'Skipping rope' },
      { icon: 'üè∏', name: 'Raket' },
      { icon: 'üèãÔ∏è', name: 'Dumbbell' },
      { icon: 'üßò', name: 'Matras yoga' }
    ]

    function shuffle(a) { for (let i = a.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [a[i], a[j]] = [a[j], a[i]] } }

    const imgPlaceholder = document.getElementById('imgPlaceholder')
    const imgChoices = document.getElementById('imgChoices')
    const imgResult = document.getElementById('imgResult')

    function startImageGame() {
      imgChoices.innerHTML = ''
      imgResult.textContent = ''
      // pick one target
      const pool = [...imageItems]
      shuffle(pool)
      const target = pool[0]
      imgPlaceholder.textContent = target.icon

      // prepare choices
      const opts = pool.slice(0,4)
      shuffle(opts)
      opts.forEach(o => {
        const b = document.createElement('button')
        b.className = 'btn btn-outline'
        b.textContent = o.name
        b.onclick = () => {
          if (o.name === target.name) {
            imgResult.textContent = 'Benar!'
          } else {
            imgResult.textContent = 'Salah. Jawaban: ' + target.name
          }
        }
        imgChoices.appendChild(b)
      })
    }

    startImageGame()

    // small accessibility: keyboard to navigate sections with numbers 1-5
    window.addEventListener('keydown', e => {
      if (e.key >= '1' && e.key <= '5') {
        const map = { '1':'home','2':'manfaat','3':'jenis','4':'tips','5':'games' }
        showSection(map[e.key])
      }
    })
  </script>
